create table person
(
    id_person     integer generated by default as identity
        primary key,
    fio           varchar not null,
    year_of_birth integer not null
);

alter table person
    owner to postgres;

create unique index person_fio_uindex
    on person (fio);

create table book
(
    id_book      integer generated by default as identity
        primary key,
    people_id    integer
                         references person
                             on delete set null,
    name_book    varchar not null,
    author_book  varchar not null,
    year_book    integer not null,
    date_of_take timestamp
);

alter table book
    owner to postgres;
